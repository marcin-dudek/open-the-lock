lock - pre .NET 4.0


Monitor.Enter(locker);
try
{
    // access to shared state
}
finally
{
    Monitor.Exit(locker);
}

lock - post .NET 4.0


bool lockTaken;
try
{
    Monitor.Enter(locker, out lockTaken)
    // access to shared state
}
finally
{
    if(lockTaken)
    {
        Monitor.Exit(locker);
    }            
}
